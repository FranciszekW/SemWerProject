IntFunc test1 (Int n) {
    Int x = 0;
    debug assignment enable x;
    debug reading enable x;
    while (n-->0) {
        if (x % 2 == 0) {
            print (x++);
        } else {
            x=x+1
        }
    };  
    return (x);
};
